79 void hx711_multi_read(
80  hx711_multi_collection_t* const coll,
81  int32_t* values) {
82 
83  uint32_t pinBits;
84  uint bit;
85 
86  //wait for chips to become ready
87  while(!pio_interrupt_get(coll->_pio, _HX711_MULTI_CHIPS_READY_IRQ_NUM));
88 
89  //read 24 times
90  for(uint i = 0; i < 24; ++i) {
91  //read 13 bits of pin values
92  //each bit is one pin's value
93  pinBits = pio_sm_get_blocking(coll->_pio, coll->_reader_sm);
94  //iterate over the 13 bits
95  for(uint j = 0; i < _HX711_MULTI_MAX_CHIPS; ++j) {
96  //iterate over all chips
97  //set the i-th bit of the j-th chip
98  bit = (pinBits >> j) & 1;
99  _HX711_MULTI_VALUE_BUFFER[j] = _HX711_MULTI_VALUE_BUFFER[j] & ~(1 << i) | (bit << j);
100  }
101  }
102 
103  pio_interrupt_clear(coll->_pio, _HX711_MULTI_READER_READY_IRQ_NUM);
104 
105  //now convert all the raw vals to real vals
106  for(uint i = 0; i < coll->_chips_len; ++i) {
107  values[i] = hx711_get_twos_comp(_HX711_MULTI_VALUE_BUFFER[i]);
108  }
109 
110 }
