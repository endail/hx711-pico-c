302 static inline bool hx711__try_get_value(
303  PIO const pio,
304  const uint sm,
305  uint32_t* const val) {
306 
307  assert(pio != NULL);
308  assert(val != NULL);
309 
310  static const uint byteThreshold = HX711_READ_BITS / 8;
311 
312  if(pio_sm_get_rx_fifo_level(pio, sm) >= byteThreshold) {
313  *val = pio_sm_get(pio, sm);
314  return true;
315  }
316 
317  return false;
