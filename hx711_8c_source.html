231 bool hx711_get_value_timeout(
232  hx711_t* const hx,
233  const uint timeout,
234  int32_t* const val) {
235 
236  assert(hx != NULL);
237  assert(hx->_pio != NULL);
238  assert(val != NULL);
239  assert(pio_sm_is_claimed(hx->_pio, hx->_state_mach));
240  assert(mutex_is_initialized(&hx->_mut));
241 
242  bool success = false;
243  static const uint byteThreshold = HX711_READ_BITS / 8;
244  const absolute_time_t endTime = make_timeout_time_us(timeout);
245  uint32_t tempVal;
246 
247  assert(!is_nil_time(endTime));
248 
249  mutex_enter_blocking(&hx->_mut);
250 
251  while(!time_reached(endTime)) {
252  if(pio_sm_get_rx_fifo_level(hx->_pio, hx->_state_mach) >= byteThreshold) {
253  //obtain value and relinquish the mutex ASAP
254  tempVal = pio_sm_get(hx->_pio, hx->_state_mach);
255  success = true;
256  break;
257  }
258  }
259 
260  mutex_exit(&hx->_mut);
261 
262  if(success) {
263  *val = hx711_get_twos_comp(tempVal);
264  }
265 
266  return success;
267 
268 }

270 void hx711_set_power(hx711_t* const hx, const hx711_power_t pwr) {
271 
272  assert(hx != NULL);
273  assert(hx->_pio != NULL);
274  assert(pio_sm_is_claimed(hx->_pio, hx->_state_mach));
275  assert(mutex_is_initialized(&hx->_mut));

277  mutex_enter_blocking(&hx->_mut);
278 
279  if(pwr == hx711_pwr_up) {
280 
296  gpio_put(hx->clock_pin, false);
297 
298  //2. reset the state machine using the default config
299  //obtained when init'ing.
300  pio_sm_init(

302  hx->_state_mach,
303  hx->_offset,
304  &hx->_default_config);
305 
306  //3. start the state machine
307  pio_sm_set_enabled(
308  hx->_pio,
309  hx->_state_mach,
310  true);
311 
312  }
313  else if(pwr == hx711_pwr_down) {
314 
315  //1. stop the state machine
316  pio_sm_set_enabled(
317  hx->_pio,
318  hx->_state_mach,
319  false);
320 
332  gpio_put(hx->clock_pin, true);
333 
334  }
335 
336  mutex_exit(&hx->_mut);
337
